## Usage

Use Logo to display branding consistently.

```tsx
import { Logo } from '@/components/ui-component/Foundational/Logo/Logo'

export default function Page() {
  return (
    <div className="space-y-4">
      <Logo size="lg" />
      <Logo variant="icon-only" />
      <Logo text="MyBrand" variant="text-only" />
    </div>
  )
}
```

## Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| text | string | "Animix" | Brand name text |
| variant | 'text-only' \| 'icon-only' \| 'full' | 'full' | Display mode |
| size | 'sm' \| 'md' \| 'lg' | 'md' | Overall scale |

import React from 'react';

type LogoProps = {
    icon?: React.ReactNode;
    text?: string;
    variant?: 'text-only' | 'icon-only' | 'full';
    size?: 'sm' | 'md' | 'lg';
    className?: string;
};

export const Logo = ({
    icon = (
        <svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" fill="currentColor" className="w-full h-full">
            <path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zm0 18c-4.41 0-8-3.59-8-8s3.59-8 8-8 8 3.59 8 8-3.59 8-8 8zm-1-13h2v6h-2zm0 8h2v2h-2z" fillOpacity="0" />

    ),
    text = "Animix",
    variant = 'full',
    size = 'md',
    className = ''
}: LogoProps) => {

    // Size logic
    const iconSizes = {
        sm: 'w-5 h-5',
        md: 'w-6 h-6',
        lg: 'w-8 h-8',
    };

    const textSizes = {
        sm: 'text-lg',
        md: 'text-xl',
        lg: 'text-2xl',
    };

    return (
        <div className={`flex items-center gap-2 font-bold text-neutral-900 dark:text-white ${className}`}>
            {variant !== 'text-only' && (
                <div className={`${iconSizes[size]} text-blue-600`}>
                    {icon}
                </div>
            )}
            {variant !== 'icon-only' && (
                <span className={`${textSizes[size]} tracking-tight`}>
                    {text}
                </span>
            )}
        </div>
    );
};
