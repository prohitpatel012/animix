## Usage

Use Image for consistently styled visual content.

```tsx
import { Image } from '@/components/ui-component/Foundational/Image/Image'

export default function Page() {
  return (
    <div className="grid grid-cols-2 gap-4 max-w-md">
      <Image 
        src="https://images.unsplash.com/photo-1682687220742-aba13b6e50ba" 
        alt="Landscape" 
        aspectRatio="video" 
      />
      <Image 
        src="https://images.unsplash.com/photo-1682687220063-4742bd7fd538" 
        alt="Square" 
        aspectRatio="square" 
        radius="full"
      />
    </div>
  )
}
```

## Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| src | string | - | Image source URL |
| alt | string | - | Alt text for accessibility |
| aspectRatio | 'auto' \| 'square' \| 'video' \| 'portrait' | 'auto' | Aspect ratio of the image container |
| fit | 'cover' \| 'contain' \| 'fill' | 'cover' | Object fit behavior |
| radius | 'none' \| 'sm' \| 'md' \| 'lg' \| 'full' | 'md' | Border radius |

## Source Code

```tsx
import React from 'react';

type ImageProps = {
    src: string;
    alt: string;
    aspectRatio?: 'auto' | 'square' | 'video' | 'portrait';
    fit?: 'cover' | 'contain' | 'fill' | 'none' | 'scale-down';
    radius?: 'none' | 'sm' | 'md' | 'lg' | 'full';
    className?: string;
};

const aspectRatios = {
    auto: 'aspect-auto',
    square: 'aspect-square',
    video: 'aspect-video',
    portrait: 'aspect-[3/4]',
};

const fits = {
    cover: 'object-cover',
    contain: 'object-contain',
    fill: 'object-fill',
    none: 'object-none',
    'scale-down': 'object-scale-down',
};

const radii = {
    none: 'rounded-none',
    sm: 'rounded-sm',
    md: 'rounded-md',
    lg: 'rounded-lg',
    full: 'rounded-full',
};

export const Image = ({
    src,
    alt,
    aspectRatio = 'auto',
    fit = 'cover',
    radius = 'md',
    className = ''
}: ImageProps) => {
    return (
        <div className={`overflow-hidden bg-neutral-100 dark:bg-neutral-800 ${radii[radius]} ${aspectRatios[aspectRatio]} ${className}`}>
            <img
                src={src}
                alt={alt}
                className={`w-full h-full ${fits[fit]}`}
            />
        </div>
    );
};
```
