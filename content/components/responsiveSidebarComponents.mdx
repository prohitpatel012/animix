## Usage

Use the sidebar for main application navigation.

```tsx
import Sidebar from '@/components/ui-component/ResponsiveSidebar/ResponsiveSidebar'

export default function Layout() {
  return (
    <div className="flex">
        <Sidebar />
        <main>Content</main>
    </div>
  )
}
```
