# Stats & Metrics

Numbers speak louder than words. Use these components to validate your claims, show scale, and demonstrate growth.

## Simple
Clean, large numbers.

<div className="my-8 border rounded-lg overflow-hidden">
  <StatsSimple />
</div>

```tsx
"use client";

const stats = [
    { id: 1, name: 'Transactions every 24 hours', value: '44 million' },
    { id: 2, name: 'Assets under holding', value: '$119 trillion' },
    { id: 3, name: 'New users annually', value: '46,000' },
]

export default function StatsSimple() {
    return (
        <div className="bg-white dark:bg-neutral-950 py-24 sm:py-32">
            <div className="mx-auto max-w-7xl px-6 lg:px-8">
                <dl className="grid grid-cols-1 gap-x-8 gap-y-16 text-center lg:grid-cols-3">
                    {stats.map((stat) => (
                        <div key={stat.id} className="mx-auto flex max-w-xs flex-col gap-y-4">
                            <dt className="text-base leading-7 text-neutral-600 dark:text-neutral-400">{stat.name}</dt>
                            <dd className="order-first text-3xl font-semibold tracking-tight text-neutral-900 dark:text-white sm:text-5xl">
                                {stat.value}
                            </dd>
                        </div>
                    ))}
                </dl>
            </div>
        </div>
    )
}
```
